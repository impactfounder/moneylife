## 💡 계산기 개발 패턴

모든 계산기는 동일한 구조를 따릅니다:

```typescript
// 1. 상태 관리
const [input1, setInput1] = useState('')
const [result, setResult] = useState(null)

// 2. 계산 함수
const handleCalculate = (e: React.FormEvent) => {
  e.preventDefault()
  const calcResult = calculateSomething(input1, input2)
  setResult(calcResult)
}

// 3. UI 구조
return (
  <>
    <Header />
    <main>
      {/* 입력 폼 */}
      <Card>
        <form onSubmit={handleCalculate}>
          <Input ... />
          <Button type="submit">계산하기</Button>
        </form>
      </Card>
      
      {/* 결과 표시 */}
      {result && (
        <Card>
          <h2>{result.value}</h2>
        </Card>
      )}
    </main>
    <Footer />
  </>
)
```
